常规快速排序

```c++
void quick_sort(vector<int>& arr,int left,int right){
    if(left >= right)   return;
    int l = left, r = right, pivot = arr[left];
    while(l < r){
        while(l < r && arr[r] >= pivot) --r;
        arr[l] = arr[r];
        while(l < r && arr[l] <= pivot) ++l;
        arr[r] = arr[l];
    }
    arr[l] = pivot;
    quick_sort(arr,left,l-1);
    quick_sort(arr,r+1,right);
}
```

```c++
void merge_sort(vector<int>& arr,int l,int r){
    if(l >= r)   return;

    int mid = (l + r) / 2;
    merge_sort(arr,l,mid);
    merge_sort(arr,mid+1,r);

    int i = l, j = mid + 1, k = 0;
    while(i <= mid && j <= r){
        if(arr[i] <= arr[j])    tmp[k++] = arr[i++];
        else    tmp[k++] = arr[j++];

    while(j <= r) tmp[k++] = arr[j++];

    for(i = l, k = 0; i <= r; ++i, ++k)  arr[i] = tmp[k];
}
```

####	3.堆排序

常规堆排序

```c++
void hepifly(vector<int>& arr,int start, int end){
    int parent = start, child = 2 * parent + 1;
    while(child <= end){
        if(child + 1 <= end && arr[child + 1] > arr[child])
            ++child;
        if(arr[parent] > arr[child])    return;
        else{
            swap(arr[parent],arr[child]);
            parent = child;
            child = 2 * parent + 1;
        }
    }
}

void heap_sort(vector<int>& arr,int n){
    for(int i = (n-2) / 2; i >= 0; --i)    hepifly(arr,i,n-1);
    for(int i = n - 1; i > 0; --i){
        swap(arr[i],arr[0]);
        hepifly(arr,0,i-1);
    }
}
```

```c++
void quick3way(vector<int>& arr,int left,int right){
    if(left >= right)   return;
    int pivot = arr[left + rand() % (right - left + 1)];
    int i = left, l = left, r = right;
    while(i <= r){
        if(arr[i] < pivot)  swap(arr[l++],arr[i++]);
        else if(arr[i] > pivot) swap(arr[i],arr[r--]);
        else ++i;
    }
    // arr 在 [l, r] 上等于 pivot
    quick3way(arr,left,l-1);
    quick3way(arr,r+1,right);
}
```

```c++
int bin_search(vector<int>& arr,int target){
    int l = 0, r = arr.size() - 1;
    while(l <= r){
        int mid = l + r >> 1;
        if(arr[mid] > target) r = l - 1;
        else    l = mid + 1;
    }
    //循环不变量	: arr[l - 1]始终小于k , arr[r + 1]始终大于等于k

}
```

###	递归实现枚举

####	1.指数型枚举 

从1~n这n（n < 20）个整数中随机选取任意多个，输出所有可能的选择方案

```c++
vector<vector<int>> ans;
void solve(vector<int>& arr,int i){
    if(i > n){
        ans.push_back(arr);
        return;
    }
    // 不选
    solve(arr,i+1);
    
    // 选
    arr.push_back(i);
    solve(arr,i+1);
    arr.pop_back();		// 回溯时还原现场
}
```

这就是所谓的剪枝，如果能及时确定当前问题一定是无解的，就不需要到达问题边界才返回结果。剪枝保证我们一旦进入无解的分支就会立刻返回，所以时间复杂度进一步降低



####	3.排列型枚举 n!

把1~n这n（n<20）个整数排成一行后随机打乱顺序，输出所有可能的次序

```c++
// 排列型枚举
vector<bool> vis(n+1,false);
vector<vector<int>> ans;	
// 函数签名也可以是这样 solve3(...arr,int cnt)
void solve3(vector<int>& arr){	
	if(arr.size() == n){
		ans.push_back(arr);
		return;
	}
	for(int i = 1; i <= n; ++i) if(!vis[i]){
		arr.push_back(i);
		vis[i] = true;
		solve3(arr);
		vis[i] = false;
		arr.pop_back();
	}	
}
```

##	数据结构



###	并查集

并查集支持的核心操作：合并，查询

下面是维护子集 ***元素个数*** 和 ***元素值的总和*** 信息的并查集，且包括移动和删除操作的模板

```c++
struct DSU{
	vector<int> pa, size, sum;
    
    // 为什么开两倍的 pa, 主要是为了可以删除节点。
    // [0,sz-1] 区间是实际的节点, 它们的父节点是虚拟节点 [sz,2*sz-1]。这样保证了每个节点都是叶子节点, 可以被删除
    explicit DSU(int sz) : pa(sz*2), size(sz*2,1), sum(sz*2){
        iota(pa.begin(),pa.begin()+sz,sz);
        iota(pa.begin()+sz,pa.end(),sz);
        iota(sum.begin()+sz,sum.end(),0);
    }
    
    int find(int x){
        return pa[x] == x ? x : pa[x] = find(pa[x]);
    }
    // x --> y
    void unite(int x,int y){
        int fx = find(x), fy = find(y);
        if(fx == fy)	return;
        if(size[fx] > size[fy])	swap(fx,fy);
        size[fy] += size[fx];
        pa[fx] = fy;
        sum[fy] += sum[fx];
    }
    
    void erase(int x){
        --size[find(x)];
        pa[x] = x;
    }
   
    // x --> y
    void move(int x,int y){
        int fx = find(x), fy = find(y);
        if(fx == fy)	return;
        sum[fx] -= x;	sum[fy] += x;
        size[fx]--;	size[fy]++;
        pa[x] = fy;
    }

    int getSum(int x){
        return sum[find(x)];
    }

    int getSize(int x){
        return size[find(x)];
    }
};
```

###	单源最短路径（不带负权）

$ n = 点的数量, m = 边的数量 $

$ 如果是稠密图, 则m和n^2是一个级别$

####	1.朴素 Dijkstra

$ 时间复杂度 O (n^2)$

```c++
vector<vector<int>> g;
vector<int> dis(maxn,0x3f3f3f3f), vis(maxn,0x3f3f3f3f);
void dijkstra(int n,int s){
    dis[s] = 0;
    // 循环 n 次或 n-1 次都可以,如果循环 n-1, 则 vis 中有一个点未被标记
    for(int i = 1; i <= n; ++i){	
        int u = 0, minDis = INT_MAX;
        // 找当前未访问过且距离最小的节点, 这一步可以用堆优化
        for(int j = 0; j < n; ++j)
            if(!vis[j] && dis[j] < minDis)	u = j, minDis = dis[j];
        vis[u] = true;
        for(int v = 0; v < n; ++v){
            if(dis[v] > dis[u] + g[u][v])	dis[v] = dis[u] + g[u][v]
        }
    }
}
```





####	2.堆优化 Dijkstra

$ 时间复杂度O(mlogm)$

```c++
using PII = pair<int,int>;
vector<vector<int>> g;
vector<int> dis(maxn,0x3f3f3f3f), vis(maxn,0);
void dijkstra(int n,int s){
    dis[s] = 0;
    priority_queue<PII,vector<PII>,greater<PII>> heap;
    heap.push({0, s});
    while(!heap.empty()){
        auto [w, u] = heap.top();	heap.pop();
        if(vis[u])	continue;
        vis[u] = true;
        for(int v = 0; v < n; ++v){
            if(g[u][v] + w < dis[v]){
                dis[v] = g[u][v] + w;
                heap.push({dis[v], v});
            }
        }
    }
}
```



稠密图可以使用朴素版本，稀疏图可以使用堆优化版

```c++
vector<int> dis;
```

```c++
bool spfa(int n,int s){
    dis[s] = 0;
    queue<int> q;
    q.push(s);
    vis[s] = true;
    while(!q.empty()){
        int u = q.front();	q.pop();
        vis[u] = false;
        for(int v = 0; v < n; ++v)	if(dis[v] > dis[u] + g[u][v]){
        	dis[v] = dis[u] + g[u][v];

    }
}
```

```c++
using PII = pair<int,int>;
const int INF = 0x3f3f3f3f;
vector<vector<int>> G(N,vector<int>(N,INF));
vector<bool> vis(N);
vector<int> dis(N,INF);
int cnt = 0;
int prim(int s){
    priority_queue<PII,vector<PII>,greater<PII>> heap;
    int res = 0;
    dis[s] = 0;
    heap.emplace(0,s);
    while(!heap.empty()){
        if(cnt >= N)	break;
        auto [w, u] = heap.top();	heap.pop();
        if(vis[u])	continue;
        vis[u] = true;
        ++cnt;
        res += dis[u];
        for(int v = 0; v < N; ++v)	if(!vis[v] && dis[v] > G[u][v]){
            dis[v] = G[u][v];
            heap.push(dis[v],v);
        }
    }
}
```

####	3.Kruskal

并查集

```c++
struct Edge{
    int u, v, w;
    Edge(int uu = 0,int vv = 0,int ww = 0) : u(uu), v(vv), w(ww) {}
    bool operator<(const Edge& rhs) const{
        return w < rhs.w;
    }
};

int n, m;
int pa[N];
vector<Edge> edges;

int find(int x){
    return pa[x] == x ? x : pa[x] = find(x);
}

int kruskal(){
    sort(edges.begin(),edges.end());
    for (int i = 1; i <= n; i ++ ) p[i] = i;    // 初始化并查集
    int res = 0, cnt = 0;
    for(int i = 0; i < m; ++i){
        int u = edges[i].u, v = edges[i].v, w = edges[i].w;
        if(find(u) != find(v)){
            pa[find(v)] = find(u);
            res += w;
            ++cnt;
        }
    }
    
    if(cnt < n - 1)	return INF;
    return res;
}
```
